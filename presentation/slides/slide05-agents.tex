% Slide 05 — Agentes Inteligentes
\begin{frame}{Agentes Inteligentes}
  \begin{block}{Definição (Wooldridge \& Jennings, 1995)}
    Entidade computacional autônoma capaz de perceber seu ambiente e
    agir sobre ele de forma a alcançar seus objetivos.
  \end{block}

  \vspace{0.5em}

  \begin{columns}[T]
    \begin{column}{0.45\textwidth}
      \textbf{Propriedades Fundamentais}
      \begin{itemize}
        \item \textbf{Autonomia} --- opera sem intervenção direta
        \item \textbf{Reatividade} --- percebe e responde ao ambiente
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \phantom{.}
      \begin{itemize}
        \item \textbf{Proatividade} --- inicia comportamentos dirigidos a objetivos
        \item \textbf{Sociabilidade} --- interage com outros agentes
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
