% Slide 06 — Sistemas Multi-Agentes
\begin{frame}{Sistemas Multi-Agentes}
  \begin{columns}[T]
    \begin{column}{0.47\textwidth}
      \begin{block}{Conceitos Fundamentais}
        Múltiplos agentes interagindo em ambiente compartilhado
        (Weiss, 1999; Russell \& Norvig, 2021)
      \end{block}
      \vspace{0.3em}
      \begin{description}
        \item[Coordenação] Organização de ações entre agentes
        \item[Comunicação] Troca de informações e intenções
        \item[Cooperação] Trabalho conjunto para objetivos comuns
      \end{description}
    \end{column}
    \begin{column}{0.47\textwidth}
      \begin{block}{Frameworks Colaborativos}
        \begin{itemize}
          \item \textbf{ChatDev} (Qian et al., 2024)\\
                Agentes LLM com papéis predefinidos em desenvolvimento de software
          \item \textbf{CoELA} (Zhang et al., 2023)\\
                Agentes corporificados com LLMs para raciocínio e linguagem
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
