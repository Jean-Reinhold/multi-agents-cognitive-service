% Slide 07 — Arquiteturas Cognitivas e LLMs
\begin{frame}{Arquiteturas Cognitivas e LLMs}
  \begin{columns}[T]
    \begin{column}{0.47\textwidth}
      \begin{block}{CoALA (Sumers et al., 2024)}
        Arquiteturas cognitivas para agentes de linguagem:
        \begin{itemize}
          \item \textbf{Memória:} trabalho, episódica, semântica, procedural
          \item \textbf{Espaço de ação}
          \item \textbf{Tomada de decisão}
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.47\textwidth}
      \begin{block}{Wang et al. (2024)}
        Framework unificado com quatro módulos:
        \begin{itemize}
          \item \textit{Profile}
          \item \textit{Memory}
          \item \textit{Planning}
          \item \textit{Action}
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

  \begin{exampleblock}{Base Teórica}
    Ambos fornecem fundamentos para agentes mais adaptativos e conscientes de memória.
  \end{exampleblock}
\end{frame}
