\begin{frame}{Pontos Fortes da Arquitetura}
  \begin{exampleblock}{Contribuições Relevantes}
    \begin{itemize}
      \item \textbf{Modularidade e escalabilidade:} abordagem
            \textit{task-driven} permite adicionar novos agentes sem
            modificar a estrutura existente
      \item \textbf{Memória compartilhada (\textit{blackboard}):} facilita
            troca de informações sem comunicação direta
      \item \textbf{Integração de múltiplos tipos de memória:} Working,
            Semantic (RAG) e Procedural, alinhada ao CoALA
      \item \textbf{Paradigma conversacional \textit{Report}:} coordenação
            flexível entre orquestrador e agentes
      \item \textbf{Uso de personas:} especialização comportamental dos agentes
    \end{itemize}
  \end{exampleblock}
\end{frame}
