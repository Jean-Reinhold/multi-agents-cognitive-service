\begin{frame}{Limitações Identificadas}
  \begin{alertblock}{Desafios e Lacunas}
    \begin{enumerate}
      \item \textbf{Avaliação limitada:} apenas um cenário de aplicação,
            dificultando a generalização dos resultados
      \item \textbf{Memória compartilhada sem controle:} ausência de
            mecanismos de resolução de conflitos e consistência
      \item \textbf{Comunicação centralizada:} interação entre agentes
            ocorre exclusivamente via orquestrador
      \item \textbf{Dependência de modelo único:} avaliação realizada
            apenas com GPT-4o
      \item \textbf{Falta de \textit{benchmarks}:} sem comparação com
            métricas padronizadas da literatura
    \end{enumerate}
  \end{alertblock}
\end{frame}
